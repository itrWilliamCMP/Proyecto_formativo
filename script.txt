CREATE TABLE USUARIOS2ProyectoFormativo(
UUID_Usuario VARCHAR2(50) PRIMARY KEY,
correoElectronico VARCHAR2(100) NOT NULL UNIQUE,
contrasena VARCHAR2(255) NOT NULL
);


CREATE TABLE PACIENTES (
id_paciente NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
nombres VARCHAR2(100),
apellidos VARCHAR2(100),
edad INT,
enfermedad VARCHAR2(150)
);


CREATE TABLE HABITACIONES (
id_habitacion NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
id_paciente number,
numero_habitacion INT,
numero_cama INT,
CONSTRAINT FK_id_paciente_habitacion FOREIGN KEY (id_paciente) REFERENCES PACIENTES(id_paciente)
);


CREATE TABLE MEDICAMENTOS (
id_medicamento NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
medicamento_asignado VARCHAR2(100),
hora_aplicacion VARCHAR2(150)
);


CREATE TABLE PACIENTES_MEDICAMENTOS (
id_pacientemedicamento NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
id_paciente number,
id_medicamento number,
CONSTRAINT FK_id_paciente_medicamento FOREIGN KEY (id_paciente) REFERENCES PACIENTES(id_paciente),
CONSTRAINT FK_id_medicamento_paciente FOREIGN KEY (id_medicamento) REFERENCES MEDICAMENTOS(id_medicamento)
);
